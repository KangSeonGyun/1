```html

<!DOCTYPE html>
<!-- !를 입력면 vscode가 자동으로 만들어 준다 -->

<!-- html문서를 파일 입출력으로 열어보는 방법(주소표시줄에 파일 경로가 뜸)은 현재 사용자의 권한을 물려받고 실행되기에 보안상 좋지 않다 -->
<!-- http프로토콜을 통해 원격에서 문서를 넘겨받아서 실행해야 한다 -->
<!-- 브라우저는 웹으로 넘겨받은 문서를 열게 될 경우 시스템에 있는 어떠한 것도 접근하지 못하게 보안 모드가 설정되어 있다 -->
<!-- 따라서 VsCode 확장프로그램인 Live Server설치해 이용한다 -->
<!-- 파일 우클릭 > Open with Live Server (단축키 Alt+L Alt+O) -->

<html lang="en">

<head>
    <!-- 컨텐츠 or 데이터를 위한 데이터 / 메타데이터를 갖고있다 -->
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>

<body>
    <!-- 방을 만들어주는 태그 / 자기 크기를 같고 있는 블럭태그 ex)h4, p와 같은 대부분의 태그 -->
    <!-- 컨텐츠를 설명하는 태그 / 인라인태그-->

    <h>부안에 오면 꼭! 들리는 카페</h> <!-- 카페 전경 이미지를 설명해주기 때문에 제목이다 -->
    
    
    <div>전체메뉴</div> <!-- 정답은 없지만 문장태그<p>는 서술어가 없기에 아니고 제목태그<hn>도 아니다 -->

    딸기청
    알랜드 아메리카노
    버터쿠키

    <div>알랜드 오시는 길</div>

    <!-- ----------------------------------------------------------------------------- -->
    
    <div> <!-- 모든 태그는 구분자다 division은 범용적인 도구 -->
        <div>과일청</div>

        <h4>직접 만든 과일청을 맛보세요.</h4>

        <p> <!-- paragraph  -->
            신선한 과일과 알랜드만의 레시피로
            과일향의 풍미를 충분히 느낄 수 있는
            수제청을 드셔보세요.
        </p>
    </div>

    <div>
        <div>쿠키</div>

        <h4>직접 구운 수제 쿠키를 만나보세요</h4>

        <p>
            신선한 버터 그리고 견과류를
            이용해 바삭함을 더해 매일마다
            직접 구운 맛난 쿠키를 만나보세요
        </p>
    </div>

    <div>
        <div>커피</div>

        <h4>다양한 로스팅으로 다채로운 맛을 느껴보세요</h4>

        <p>
            콜롬비아산의 상큼한 맛, 과테말라산의 풍미,
            그 외에 5가지 이상의 다채로운 원두의 맛을 느껴보세요.
        </p>
    </div>

    <div>
        <div>명소</div>

        <h4> 알랜드 주변의 명소를 찾아보세요.</h4>

        <p>
            알랜드 주변에는 30곳이 넘는 힐링 장소에서 맛나는 커피와 경치로 힐링을 해보세요.
        </p>
    </div>
</body>

</html>

```
