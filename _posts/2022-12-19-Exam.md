```java

package ex1;

public class Exam {
	int kor;
	int eng;
	int math;

	public void input() {
//		주로 쓰는 함수 this를 사용할 수 있다

	}

	public void print() {
		int kor = this.kor;

		System.out.printf("kor:%d\n", kor);
		System.out.printf("eng:%d\n", this.eng);
		System.out.printf("math:%d\n", this.math);

	}
}

```
