```java


// 접근 제한자 종류
// public, protected, default, private / 오른쪽으로 갈 수록 접근 제한 강화

//  접근 제한자  /          적용 대상          /        접근 가능 대상       / 접근 불가 대상
// public       / 클래스, 필드, 생성자, 메소드 /          모든 대상         / 없음
// protected    / 필드, 생성자, 메소드        /  같은 패키지 or 자식 클래스  / 자식 클래스가 아닌 다른 패키지에 소속된 클래스
// default      / 클래스, 필드, 생성자, 메소드 / 같은 패키지에 소속된 클래스 / 다른 패키지에 소속된 클래스
// private      / 필드, 생성자, 메소드        /         클래스 내부         / 클래스 외부

//  접근 제한자   / 클래스 내부 / 동일 패키지 / 상속받은 클래스 / 이외의 영역
// private	 /      O      /      X      /        X        /    X
// default	 /      O      /      O      /        X        /    X
// protected     /      O      /      O      /        O        /    X
// public	 /      O      /      O      /        O        /    O

// 클래스, 필드, 생성자, 메소드 선언 시 접근 제한자를 생략한다면 default 접근 제한을 가진다

// ---------------------------------------------------------------------------------

// 1. Class 접근 제한

public class Program {
}
// 같은 패키지에서는 제약 없이 사용 가능 / 다른 패키지에서도 사용 가능
// 다른 개발자가 클래스를 사용하도록 라이브러리 클래스로 개발 시, public 접근 제한을 갖게 한다

class Program2 {
}
// 같은 패키지에서는 제약 없이 사용 가능 / 다른 패키지에서는 사용 불가

// ---------------------------------------------------------------------------------

// 2. Constructor 접근 제한
// 생성자를 선언하지 않아 자동으로 생성되는 기본 생성자의 접근 제한은 클래스의 접근 제한과 동일하다

class ClassName {
	public ClassName(int a) {
//	모든 패키지에서 제한 없이 생성자 호출 가능
//	클래스가 default 접근 제한을 가질 때 클래스 사용이 같은 패키지로 한정되므로 생성자가 public 접근 제한을 가지더라도 같은 패키지 내에서만 생성자 호출 가능
//	생성자가 public 접근 제한이면 클래스도 public 접근 제한을 가지는 것이 정상적

	}

	protected ClassName(int a, int b) {
//	같은 패키지에 속하는 클래스에서 생성자 호출 가능
//	다른 패키지에 속한 클래스가 해당 클래스의 자식 클래스라면 생성자를 호출할 수 있다
	}

	ClassName() {
//	같은 패키지 내에서 아무런 제한 없이 생성자 호출 가능
//	다른 패키지에서는 생성자를 호출할 수 없다
	}

	private ClassName(int a, int b, int c) {
//	생성자를 호출할 수 없다 / 클래스 외부에서 new 연산자로 객체 생성 불가
//	클래스 내부에서만 생성자를 호출할 수 있고, 객체를 만들 수 있다
	}
}

//---------------------------------------------------------------------------------

// 3. 필드와 메소드 접근 제한
// static으로도 선언 가능하다

class Program3 {
	public int field; // 접근 제한자 타입 필드;
	public void method() {} // 접근 제한자 리턴타입 필드;
//	모든 패키지에서 아무 제한 없이 필드, 메소드 사용가능
//	필드, 메소드가 public 접근 제한일 경우 클래스도 public 접근 제한을 가져야 한다
//	클래스가 default 접근 제한이라면, 같은 패키지 안에서만 클래스가 사용되기 때문이다

	protected int field2;
	protected void method2() {}
//	같은 패키지에 속하는 쿨래스에서 필드, 메소드 사용가능
//	다른 패키지에 속한 클래스가 해당 클래스의 자식 클래스일 경우 필드와 메소드를 사용할 수 있다

	int field3;
	void method3() {}
//	같은 패키지에서만 필드, 메소드 사용 가능
//	다른 패키지에선 필드, 메소드 사용 불가

	private int field4;
	private void method4() {}
//	동일 패키지, 다른 패키지에서 필드, 메소드 사용 불가
//	클래스 내부에서만 사용 가능

}
```
