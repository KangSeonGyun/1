boy.js
-------------

```js

function Boy(x, y) {
    this.ix = 0;
    this.iy = 2;
    this.x = x || 100;
    this.y = y || 100;

    this.sw = 106;
    this.sh = 148.25;
    this.sx = this.sw * this.ix;
    this.sy = this.sh * this.iy;

}

Boy.prototype = {
    draw: function (ctx) {
        var img = new Image();
        img.src = "./images/boy.png";

        img.onload = function () {
            ctx.drawImage(img, this.sx, this.sy, this.sw, this.sh,
                this.x, this.y, this.sw, this.sh);
        }.bind(this);
    },
    move: function (dir) {
        switch (dir) {
            case 1: // 북
                this.y -= 10;
                break;
            case 2: // 동
                this.x += 10;
                break;
            case 3: // 남
                this.y += 10;
                break;
            case 4: // 서
                this.x -= 10;
                break;
        }
    }
}

```

game-canvas.js
-------------

```js

function GameCanvas() {

    this.dom = document.querySelector(".game-canvas");
    // canvas파일안에 dom

    /** @type {CanvasRenderingContext2D} */
    this.ctx = this.dom.getContext("2d");

    this.boy = new Boy(100, 100);

    this.dom.onclick = this.clickHandler.bind(this);
    // 델리게이션 펑션, 콜백 펑션(나중에 실행되는 함수, 사용자가 콜해야 실행한다) this.clickHandler.bind(this); 나중에 콜해줘 dom(main)이 콜 해준다
    // this.dom.onclick = this.clickHandler;과 this.dom.onclick = this.clickHandler.bind(this); 차이??
    // this는 app.js의 gameCanvas
    // dom은 main의 canvas / java는 is-a로 쓰지만 js는 is-a가 없기에 틀로써야 적절하지만 캔바스가 부품으로 캔바스를 갖고 있다
}

GameCanvas.prototype = {
    run: function () {
        // 60 프레임으로 화면을 다시 그리는 코드
        this.update();
        this.draw();
    },
    update: function () {

    },
    draw: function () {
        this.boy.draw(this.ctx);
    },

    // -----event handlers-----

    clickHandler: function () {
        this.boy.move(2);
        this.boy.draw(this.ctx);
        // 일단 dom에 boy가 없다 this로 GameCanvas.boy를 가르켜 준다
    }

}

```
