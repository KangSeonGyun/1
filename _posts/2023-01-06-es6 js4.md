    }

    } 
}

// 5. getters/setters를 사용하는 것이 편해졌다.

class Enemy{
    #x;
    #y;
    constructor(x, y){
        this.#x = x || 0;
        this.#y = y || 0;
    }
    
    // ES5
    getX(){
        return this.#x;
    }
    setX(x){
        this.#x = x;
    }  
}

// 위와 같이 setX()와 getX()를 추가하면 그것을 사용하는 방법은 다음과 같다.

// enemy=30;

enemy.setX(enemy.getX()+1);

// 하지만 이제는 getters와 setters를 다음처럼 정의할 수 있다.

class Enemy{
    #x;
    #y;
    constructor(x, y){
        this.#x = x || 0;
        this.#y = y || 0;
    }
    
    // ES6
    get x(){
        return this.#x;
    }
    set x(x){
        this.#x = x;
    }
}

// 그러면 연산이 다음처럼 간단해진다.
