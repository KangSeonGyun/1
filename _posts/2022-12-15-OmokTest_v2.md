package ex1.func;

import java.util.Scanner;
import java.io.IOException;

public class OmokTest_v2 {
	public static void main(String[] args) throws IOException {

		char[][] board = new char[10][10]; // board[y][x]

		for (int y = 0; y < 10; y++)
			for (int x = 0; x < 10; x++)
				board[y][x] = '┼';

		for (int y = 0; y < 10; y++) {
			System.out.print(y);
			for (int x = 0; x < 10; x++)
				System.out.print(board[y][x]);
			System.out.println();
		}

		System.out.print("└");
		for (int i = 0; i < 10; i++) {

		}
