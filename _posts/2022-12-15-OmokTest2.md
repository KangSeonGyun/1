		for (int y = 0; y < 10; y++)
			for (int x = 0; x < 10; x++)
				board[y][x] = '┼';

		for (int i = 0; i < 10; i++) {
			board[0][i] = '┬';
			board[9][i] = '┴';
			board[i][0] = '├';
			board[i][9] = '┤';
		}

		board[0][0] = '┌';
		board[0][9] = '┐';
		board[9][0] = '└';
